git config --global user.name "your name"  /配置用户名
git config --global user.email "email@excample.com" /配置邮箱

git init 初始化  /把该目录变成git可以管理的仓库，该目录下多数了.git文件

git add . 全部提交到缓存区
git add main.py  /该文件提交到缓存区
git commit -m /提交到本地仓库，可以一次提交所有文件
git commit -m "功能已完成" /备注修改了哪些内容


git log   查看过往记录
git status 查看工作区状态
git pull 拉去代码
git push 提交到远程仓库
git checkout HEAD 
git checkout branch_name 检查分支
git clone xxx 下载
